const AWS = require('aws-sdk'); const { v4: uuidv4 } = require('uuid'); exports.handler = async (event, context) => { try { // Configurar AWS S3 const s3 = new AWS.S3({ accessKeyId: process.env.AWS_ACCESS_KEY_ID, secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY, region: process.env.AWS_REGION }); // Parsear los datos del formulario const formData = event.body; const file = event.files && event.files[0]; const folderName = event.body && event.body.folderName; if (!file) { throw new Error('No file uploaded'); } // Parámetros para la subida a S3 const params = { Bucket: process.env.S3_BUCKET_NAME, Key: `${folderName}/${file.originalFilename || uuidv4()}.xlsx`, // Guardar en la carpeta con nombre mes-año Body: file.content, ContentType: file.contentType }; // Subir el archivo a S3 const data = await s3.upload(params).promise(); return { statusCode: 200, body: JSON.stringify({ message: 'File uploaded successfully', location: data.Location }) }; } catch (err) { return { statusCode: 500, body: JSON.stringify({ error: err.message }) }; } };
